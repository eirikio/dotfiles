# Commands:

NB: Mange av kommandoene kjøres enten direkte fra mappa du er inni eller kan kjøres om du skriver inn hele pathen til filen eller mappa.

cd C:/Users/eirik/Desktop

New-Item file.txt

eller

New-Item C:/Users/eirik/Desktop/file.txt

# Basics:

Change directory

cd

List files and folders

ls

Show current working directory

pwd

Show file contents

cat file.txt

Clear terminal screen

clear

Print something to the terminal

echo message

# Endre profiler/config i PowerShell og WSL

PowerShell:

notepad $PROFILE

Reload:

$PROFILE .

WSL - Åpne .zshrc editoren

nano ~/.zshrc

Reload:

source ~/.zshrc

# Opprette nye filer

PowerShell:

New-Item file.txt

WSL:

touch file.txt

# Endre navn på filer/mapper:

PowerShell:

Rename-Item "oldname.txt" "newname.txt"

WSL:

mv oldname.txt newname.txt

# Flytte filer & mapper

## Flytte filer:

PowerShell:

Move-Item "file.txt" "C:\Users\eirik\destination\""

WSL:

mv file.txt destination/

## Flytte mapper:

PowerShell:

Move-Item ".\foldername" "C:\Users\eirik\destination\"

WSL:

mv /path/to/foldername /path/to/destination/

# Slette filer & mapper

## Slette filer:

PowerShell:

Remove-Item filename.txt

WSL:

rm filename.txt

## Slette mapper:

PowerShell:

Remove-Item foldername -Recurse

WSL:

rm -r foldername/

# Sjekke alle Python instanser som kjører:

PowerShell:

PowerShell Get-CimInstance Win32_Process | Where-Object { $_.Name -eq "python.exe" } | Select-Object ProcessId, CommandLine

WSL:

ps aux | grep python

# Stoppe en prosess:

PowerShell:

powershell Stop-Process "PID"

WSL:

kill "PID"
